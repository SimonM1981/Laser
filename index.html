const SPREADSHEET_NAME = "Laser Data"; const SHEET_NAME = "Sheet1"; function getAllData() { const ss = SpreadsheetApp.openByName(SPREADSHEET_NAME); const sheet = ss.getSheetByName(SHEET_NAME); const values = sheet.getDataRange().getValues(); if (values.length <= 1) return []; const headers = values[0]; const data = []; for (let i = 1; i < values.length; i++) { let obj = {}; for (let j = 0; j < headers.length; j++) { obj[headers[j]] = values[i][j]; } data.push(obj); } return data; } function doGet(e) { try { if (e.parameter.action === "read") { const data = getAllData(); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({error: "Unknown action"})) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({error: err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); let rowsToAdd = Array.isArray(body[0]) ? body : [body]; if (rowsToAdd.some(row => row.length !== 11)) { throw new Error("Each row must contain exactly 11 values (B to L)"); } const ss = SpreadsheetApp.openByName(SPREADSHEET_NAME); const sheet = ss.getSheetByName(SHEET_NAME); rowsToAdd.forEach(row => { sheet.appendRow([new Date(), ...row]); }); return ContentService.createTextOutput(JSON.stringify({success: true, added: rowsToAdd.length})) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({error: err.message})) .setMimeType(ContentService.MimeType.JSON); } }
